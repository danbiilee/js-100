const nodes = '1 7'.split(' ');

function dfs(graph) {
  const visited = [nodes[0]];
  const stack = [nodes[0]];
  let count = -1;

  while (stack.length > 0) {
    for (let i = 0; i < stack.length; i++) {
      count++;
      const n = stack.pop();
      if (n === nodes[1]) {
        return count;
      }
      const sub = graph[n].filter((g) => !visited.includes(g));
      for (let s of sub) {
        visited.push(s);
        stack.push(s);
      }
    }
  }
  return count;
}
